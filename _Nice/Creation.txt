CREATE TABLE vehicules
(typeVehicule VARCHAR(30),
PRIMARY KEY (typeVehicule));

CREATE TABLE provinces
(nomProvince VARCHAR(30),
capitale VARCHAR(30),
superficie DOUBLE,
population DOUBLE,
PRIMARY KEY (nomProvince));

CREATE TABLE ventes
(Province VARCHAR(30),
TypeVeh VARCHAR(30),
Annee INT,
NbUnites DOUBLE,
Mntx1000 DOUBLE,
PRIMARY KEY (Province, TypeVeh, Annee),
FOREIGN KEY (TypeVeh) REFERENCES vehicules(typeVehicule),
FOREIGN KEY (Province) REFERENCES provinces(nomProvince));
